static T watch<T extends ChangeNotifier>(BuildContext context) {
  final inherited =
      context.dependOnInheritedWidgetOfExactType<MinInherited<T>>();
  assert(inherited != null, 'No MinProvider<$T> found in context.');
  return inherited!.notifier!;
}
